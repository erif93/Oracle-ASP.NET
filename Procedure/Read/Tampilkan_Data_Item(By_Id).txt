create or replace 
   procedure getdata(k_id in varchar2)
is

 cursor kc_data is select Name,Price,Stock from Item where 
 Id =k_id;

kc_rec kc_data%rowtype;
begin
for kc_rec in kc_data loop
 dbms_output.put_line('---------------------------------');
      dbms_output.put_line('Nama Item : ' || kc_rec.Name);
      dbms_output.put_line('Harga : ' || kc_rec.Price);
      dbms_output.put_line('Stock : ' || kc_rec.Stock);
end loop;
end;



begin
getdata(2);
end;
